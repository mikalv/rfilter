    def initialize(input, logfile)
      @logfile =
        if logfile.nil?
          nil
        else
          File.open(logfile, File::CREAT|File::APPEND|File::WRONLY, 0600)
        end
      @message = if input.is_a?(RMail::Message)
                   input
                 else
                   RMail::Parser.new.parse(input)
                 end
      @logging_level = 2
    end

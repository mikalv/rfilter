    def deliver_pipe(command, message)
      begin
        IO.popen(command, "w") { |io|
          message.each { |line|
            io << line
            io << "\n" unless line[-1] == ?\n
          }
        }
      rescue Errno::EPIPE
        # Just ignore.
      end
    end
